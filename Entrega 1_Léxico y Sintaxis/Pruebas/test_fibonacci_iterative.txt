program FibonacciIterative;
var
  n, a, b, temp, i : int;

main {
  n = 10; // Calcular Fibonacci hasta el tÃ©rmino 9 (F(0) a F(9))
  a = 0;
  b = 1;
  i = 0;

  print("Iterative Fibonacci sequence up to F(", n-1,"):");

  if (n > 0) {
    print(a);
    i = i + 1;
  };

  while (i < n) do {
    print(b);
    temp = a + b;
    a = b;
    b = temp;
    i = i + 1;
  };
}
end
