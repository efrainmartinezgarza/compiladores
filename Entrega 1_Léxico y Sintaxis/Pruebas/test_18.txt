program TestComplexStructure;
var
  i, j, k_val : int;
  f_val : float;

void NestedFunc(x : int) [
  var
    y : int;
  {
    y = x * x;
    if (y > 100) {
      print("Cuadrado grande: ", y);
    } else {
      print("Cuadrado pequenio: ", y);
    };
  }
];

main
{
  i = 0;
  f_val = 10.5;
  while (i < 3) do {
    j = i + 1;
    k_val = (j * 10) / 2;
    if (k_val != (i+1)*5) {
        print("Error en el cÃ¡lculo.");
    };
    NestedFunc(k_val - i);
    i = i + 1;
    print("Ciclo ", i, " f_val is ", f_val);
  };
  f_val = f_val * -2.0;
  print("Final f_val: ", f_val);
}
end
